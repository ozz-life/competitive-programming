n, K, D = map(int, input().split())

def f(n, K, D):
    for _ in range(D):
        n *= 10
        found = False
        for i in range(10):
            if (n + i) % K == 0:
                n += i
                found = True
                break
        if not found:
            return -1
    return n

print(f(n, K, D))
